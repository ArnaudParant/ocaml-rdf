Creating a graph

An RDF graph needs a base URI, which can be used to name the graph. Such a URI is created using the Rdf_uri module. Then we use the Rdf_graph.open_graph function to create an empty graph: 

let base_uri = Rdf_uri.uri "http://foo.bar.net";;
let g = Rdf_graph.open_graph base_uri ;;

If no option is provided, the graph is created in memory. You can create graphs stored in a MySQL or Postgresql datatabase.

Once the graph is created, let's insert triples in the graph, to associate the name "Alice" and the mailbox address URI "mailto:alice@mbox.net" to a new blank node: 

# let foaf_name = Rdf_uri.uri "http://xmlns.com/foaf/0.1/name";;
val foaf_name : Rdf_uri.uri = <abstr>
# let foaf_mbox = Rdf_uri.uri "http://xmlns.com/foaf/0.1/mbox";;
val foaf_mbox : Rdf_uri.uri = <abstr>
# let a = Rdf_node.Blank_ (g.Rdf_graph.new_blank_id());;
val a : Rdf_node.node = Rdf_node.Blank_ <abstr>
# g.Rdf_graph.add_triple
   ~sub: a ~pred: (Rdf_node.Uri foaf_name)
   ~obj: (Rdf_node.node_of_literal_string "Alice");;
- : unit = ()
# g.Rdf_graph.add_triple
   ~sub: a ~pred: (Rdf_node.Uri foaf_mbox)
   ~obj: (Rdf_node.Uri (Rdf_uri.uri "mailto:alice@mbox.net"));;
- : unit = ()
# let b = Rdf_node.Blank_ (g.Rdf_graph.new_blank_id());;
val b : Rdf_node.node = Rdf_node.Blank_ <abstr>
# g.Rdf_graph.add_triple
   ~sub: b ~pred: (Rdf_node.Uri foaf_name)
   ~obj: (Rdf_node.node_of_literal_string "Bob");;
- : unit = ()
# g.Rdf_graph.add_triple
   ~sub: b ~pred: (Rdf_node.Uri foaf_mbox)
   ~obj: (Rdf_node.Uri (Rdf_uri.uri "mailto:bob@mbox.net"));;
- : unit = ()

Here is the content of our graph, in Turtle format:

@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> .
_:genid1-1005158737 <http://xmlns.com/foaf/0.1/name> "Bob" .
_:genid1-1005158737 <http://xmlns.com/foaf/0.1/mbox> <mailto:bob@mbox.net> .
_:genid0-785609744 <http://xmlns.com/foaf/0.1/name> "Alice" .
_:genid0-785609744 <http://xmlns.com/foaf/0.1/mbox> <mailto:alice@mbox.net> .
Querying the graph

Let's use some of the Rdf_graph.graph structure fields to retrieve information from the graph.

Here we look for the mailboxes of Alice, by querying the objects of triples having a specified subject and object: 

# let alice_mbox = g.Rdf_graph.objects_of
  ~sub: a ~pred: (Rdf_node.Uri foaf_mbox);;
val alice_mbox : Rdf_node.node list = [Rdf_node.Uri <abstr>]
# List.iter (fun n -> print_endline (Rdf_node.string_of_node n)) alice_mbox;;
<mailto:alice@mbox.net>
- : unit = ()

Let's now query all triples having the predicate foaf_name, and use the Rdf_ttl.string_of_triple function to display the result: 

# let triples = g.Rdf_graph.find ~pred: (Rdf_node.Uri foaf_name) ();;
val triples : Rdf_node.triple list =
  [(Rdf_node.Blank_ <abstr>, Rdf_node.Uri <abstr>,
    Rdf_node.Literal
     {Rdf_node.lit_value = "Alice"; Rdf_node.lit_language = None;
      Rdf_node.lit_type = None});
   (Rdf_node.Blank_ <abstr>, Rdf_node.Uri <abstr>,
    Rdf_node.Literal
     {Rdf_node.lit_value = ...; Rdf_node.lit_language = ...;
      Rdf_node.lit_type = ...});
   ...]
# List.iter
   (fun (sub,pred,obj) ->
     print_endline (Rdf_ttl.string_of_triple ~sub ~pred ~obj))
   triples;;
_:genid0-785609744 <http://xmlns.com/foaf/0.1/name> "Alice" .
_:genid1-1005158737 <http://xmlns.com/foaf/0.1/name> "Bob" .
- : unit = ()

These are simple ways to query the graph. Sparql queries can be used to execute more powerful and more complex queries.

Importing and exporting

The RDF/XML and Turtle formats are supported to import and export triples. The corresponding functions are provided in modules Rdf_xml and Rdf_ttl .

Here is an example of code creating a graph, importing from a RDF/XML file example.rdf and exporting to a Turtle file example.ttl: 

# let base = Rdf_uri.uri "http://hello.fr" ;;
val base : Rdf_uri.uri = <abstr>
# let g = Rdf_graph.open_graph base ;;
val g : Rdf_graph.graph =
  {Rdf_graph.name = <fun>; Rdf_graph.add_triple = <fun>;
   Rdf_graph.rem_triple = <fun>; Rdf_graph.add_triple_t = <fun>;
   Rdf_graph.rem_triple_t = <fun>; Rdf_graph.subjects_of = <fun>;
   Rdf_graph.predicates_of = <fun>; Rdf_graph.objects_of = <fun>;
   Rdf_graph.find = <fun>; Rdf_graph.exists = <fun>;
   Rdf_graph.exists_t = <fun>; Rdf_graph.subjects = <fun>;
   Rdf_graph.predicates = <fun>; Rdf_graph.objects = <fun>;
   Rdf_graph.transaction_start = <fun>; Rdf_graph.transaction_commit = <fun>;
   Rdf_graph.transaction_rollback = <fun>; Rdf_graph.new_blank_id = <fun>;
   Rdf_graph.namespaces = <fun>}
# Rdf_xml.from_file g base "example.rdf";;
- : unit = ()
# Rdf_ttl.to_file g "example.ttl";;
- : unit = ()
Transactions

Transactions can be used if the backend support them. The storage in memory supports nested transactions, and database backends support one level of transaction.

Transactions are handled with the transaction_start, transaction_commit and transaction_rollback of the the Rdf_graph.graph structure.

As an example, let's load data from a file:

let base = Rdf_uri.uri "http://hello.fr" ;;
let g = Rdf_graph.open_graph base ;;
Rdf_ttl.from_file g base "example.ttl";;

This is our loaded data:

@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> .
@prefix dc: <http://purl.org/dc/elements/1.1/> .
@prefix ex: <http://example.org/stuff/1.0/> .
<http://www.w3.org/TR/rdf-syntax-grammar> dc:title "RDF/XML Syntax Specification (Revised)" .
<http://www.w3.org/TR/rdf-syntax-grammar> <http://hello.fr/editor> _:genid1-492098176 .
_:genid1-492098176 ex:homePage <http://purl.org/net/dajobe/> .
_:genid1-492098176 ex:fullName "Dave Beckett" .

Now let's start a transaction, and modify the graph:

g.Rdf_graph.transaction_start();;
let pred = Rdf_node.Uri (Rdf_uri.uri "http://example.org/stuff/1.0/fullName") ;;
match g.Rdf_graph.find ~pred () with
| (sub,_,obj) :: _ ->
  (* remove a triple *)
  g.Rdf_graph.rem_triple ~sub ~pred ~obj;
  (* add a new triple, with a different name *)
  g.Rdf_graph.add_triple ~sub ~pred
    ~obj: (Rdf_node.node_of_literal_string "Jean Dupont")
| _ -> assert false;;

Now the graph looks like:

@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> .
@prefix dc: <http://purl.org/dc/elements/1.1/> .
@prefix ex: <http://example.org/stuff/1.0/> .
<http://www.w3.org/TR/rdf-syntax-grammar> dc:title "RDF/XML Syntax Specification (Revised)" .
<http://www.w3.org/TR/rdf-syntax-grammar> <http://hello.fr/editor> _:genid1-492098176 .
_:genid1-492098176 ex:homePage <http://purl.org/net/dajobe/> .
_:genid1-492098176 ex:fullName "Jean Dupont" .

If we rollback, we get the previous graph:

g.Rdf_graph.transaction_rollback();;
@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> .
@prefix dc: <http://purl.org/dc/elements/1.1/> .
@prefix ex: <http://example.org/stuff/1.0/> .
<http://www.w3.org/TR/rdf-syntax-grammar> dc:title "RDF/XML Syntax Specification (Revised)" .
<http://www.w3.org/TR/rdf-syntax-grammar> <http://hello.fr/editor> _:genid1-492098176 .
_:genid1-492098176 ex:homePage <http://purl.org/net/dajobe/> .
_:genid1-492098176 ex:fullName "Dave Beckett" .
Dot output

RDF graphs can be dumped in Graphviz (dot) format, with the Rdf_dot module. The following code generate the dot code, then creates a .svg file by running a dot command. The result is inserted in below. 

let dot_file = "example.dot";;
let svg_file = (Filename.chop_extension dot_file)^".svg";;
let dot_code = Rdf_dot.dot_of_graph g;;
let () =
  let oc = open_out dot_file in
  output_string oc dot_code;
  close_out oc;;
let com = Printf.sprintf "dot -Grankdir=TB -Tsvg -o %s %s"
  (Filename.quote svg_file) (Filename.quote dot_file);;
if Sys.command com <> 0 then failwith "Exec error";;
